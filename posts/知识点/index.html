4. 泛型

4.1 最常用的泛型——集合

static T Sum<T> (List<T> nums)
    where T : struct	//泛型约束
{
    T sum = default(T);
    foreach(dynamic num in nums)
    {
        sum += num;
    }
    return sum;
}
